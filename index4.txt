const readline = require('readline');

const rl = readline.createInterface({
  input: process.stdin,
  output: process.stdout
});

rl.question('Введите сумму вклада: ', (amount) => {
  amount = parseFloat(amount);

  let interestRate;
  if (amount < 100) {
    interestRate = 0.05;
  } else if (amount >= 100 && amount <= 200) {
    interestRate = 0.07;
  } else {
    interestRate = 0.10;
  }

  const totalAmount = amount + amount * interestRate;
  console.log(Сумма вклада с начисленными процентами: ${totalAmount});
  rl.close();
});
